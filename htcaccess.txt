RewriteEngine on

# Redirect to another domain: www.anml.us.
RewriteCond %{HTTPS} off
RewriteCond %{HTTP_HOST} !^(www\.)?anml\.us$ [NC]
RewriteRule .* http://www.anml.us%{REQUEST_URI} [R=301,L]
# Same for HTTPS:
RewriteCond %{HTTPS} on
RewriteCond %{HTTP_HOST} !^(www\.)?anml\.us$ [NC]
RewriteRule .* https://www.anml.us%{REQUEST_URI} [R=301,L]

# 301 Redirect URLs.
Redirect 301 /weanml.github.io/anml2 http://www.anml.us
